import { ModalBooksAndChapters, Versions } from '../types';
import versions_withBooks from '../data/translations_books.json'


function ModalBooksAndChapters({ chapterContent, currentVersion, changeBook, changeChapter, bookSelected, chapterSelected, changeVerse }: ModalBooksAndChapters) {
    const versions: Versions = {
        ...versions_withBooks
    };
    //https://bolls.life/static/translations/YLT.json



    return (
        <div>

                <label className="modal-box relative" htmlFor="">
                    <h3 className="text-lg font-bold">Select book and chapter</h3>

                    <div className="overflow-x-auto">
                        {versions[currentVersion].map((book) => (
                            <div key={book.bookid} tabIndex={0} className="collapse collapse-arrow border border-base-300 bg-base-100 rounded-box">
                                {/* <input type="radio" defaultChecked={book.name === bookSelected.name} /> */}
                                <div className="collapse-title text-xl font-medium">
                                    <p className={`${book.name === bookSelected.name ? "text-primary" : ""}`}>{book.name}</p>
                                </div>
                                <div className="collapse-content">
                                    {Array.from({ length: book.chapters }, (_, index) => index + 1).map((number: number, indexChapter: number) => {
                                        return (
                                            <div key={indexChapter} className='collapse collapse-arrow'>
                                                {/* <label htmlFor="my-modal-books-chapters" onClick={() => changeBookandChapter(book, index + 1)} className={`btn ${(book.name === bookSelected.name && chapterSelected === index + 1) ? "btn-primary" : ""}`}>{index + 1}</label> */}

                                                    changeBook(book)
                                                    changeChapter(indexChapter + 1)
                                                }} className="collapse-title text-xl font-medium">
                                                    <p className={`${chapterSelected === indexChapter + 1 ? "text-secondary" : "text-current"}`}>{indexChapter + 1}</p>
                                                </div>
                                                <input type="radio" name="my-accordion-1" defaultChecked={chapterSelected === indexChapter + 1} />

                                                <div className="collapse-content grid grid-cols-8 gap-2">
                                                    {/* {Array.from({ length: book.chapters }, (_, index) => index + 1).map((number: number, index: number) => (
                                                            <div key={index}>
                                                                <label htmlFor="my-modal-books-chapters" onClick={() => changeBookandChapter(book, index + 1)} className={`btn ${(book.name === bookSelected.name && chapterSelected === index + 1) ? "btn-primary" : ""}`}>{index + 1}</label>
                                                            </div>
                                                        ))} */}
                                                    {
                                                        chapterContent !== undefined &&
                                                        chapterContent.map((_, indexVerse) => {
                                                            return (
                                                                <label className={`btn hover:btn-primary`} htmlFor="my-modal-books-chapters" onClick={() => changeVerse(indexVerse + 1)}>{indexVerse + 1}</label>
                                                            )
                                                        })
                                                    }
                                                </div>
                                            </div>
                                        )
                                    })}
                                </div>
                            </div>
                        ))}
                    </div>


                </label>
            </label>
        </div>
    )
}

export default ModalBooksAndChapters
